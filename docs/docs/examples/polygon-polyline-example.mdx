# Polygon & Polyline 예제

지도에 영역과 경로를 표시하는 예제입니다.

## Polygon - 영역 표시

<BrowserOnly>{() => <PolygonExample />}</BrowserOnly>

### 코드

```tsx
import { Map, Polygon } from "@rousen/react-naver-maps";

function PolygonExample() {
  const paths = [
    { x: 127.0276, y: 37.4979 },
    { x: 127.0286, y: 37.4979 },
    { x: 127.0286, y: 37.4989 },
    { x: 127.0276, y: 37.4989 },
  ];

  return (
    <div style={{ width: "100%", height: "400px" }}>
      <Map ncpKeyId="your-ncp-key-id">
        <Polygon
          paths={paths}
          fillColor="#ff0000"
          fillOpacity={0.3}
          strokeColor="#ff0000"
          strokeOpacity={0.8}
          strokeWeight={3}
        />
      </Map>
    </div>
  );
}

export default PolygonExample;
```

## Polyline - 경로 표시

<BrowserOnly>{() => <PolylineExample />}</BrowserOnly>

### 코드

```tsx
import { Map, Polyline } from "@rousen/react-naver-maps";

function PolylineExample() {
  const path = [
    { x: 127.0276, y: 37.4979 },
    { x: 127.0286, y: 37.4989 },
    { x: 127.0296, y: 37.4999 },
  ];

  return (
    <div style={{ width: "100%", height: "400px" }}>
      <Map ncpKeyId="your-ncp-key-id">
        <Polyline
          path={path}
          strokeColor="#0000ff"
          strokeOpacity={0.8}
          strokeWeight={5}
        />
      </Map>
    </div>
  );
}

export default PolylineExample;
```

## 경로 추적 시뮬레이션

```tsx
import { useState, useEffect } from "react";
import { Map, Polyline, Marker } from "@rousen/react-naver-maps";

function RouteTracking() {
  const [path, setPath] = useState<Array<{ x: number; y: number }>>([]);
  const [currentPosition, setCurrentPosition] = useState({
    x: 127.0276,
    y: 37.4979,
  });

  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentPosition((prev) => ({
        x: prev.x + (Math.random() - 0.5) * 0.001,
        y: prev.y + (Math.random() - 0.5) * 0.001,
      }));
      setPath((prev) => [...prev, { ...currentPosition }]);
    }, 1000);

    return () => clearInterval(interval);
  }, [currentPosition]);

  return (
    <div style={{ width: "100%", height: "400px" }}>
      <Map ncpKeyId="your-ncp-key-id">
        <Polyline path={path} strokeColor="#00ff00" strokeWeight={3} />
        <Marker position={currentPosition} />
      </Map>
    </div>
  );
}

export default RouteTracking;
```

## 여러 영역 표시

```tsx
import { Map, Polygon } from "@rousen/react-naver-maps";

function MultiplePolygons() {
  const areas = [
    {
      paths: [
        { x: 127.0276, y: 37.4979 },
        { x: 127.0286, y: 37.4979 },
        { x: 127.0286, y: 37.4989 },
      ],
      fillColor: "#ff0000",
    },
    {
      paths: [
        { x: 127.0296, y: 37.4999 },
        { x: 127.0306, y: 37.4999 },
        { x: 127.0306, y: 37.5009 },
      ],
      fillColor: "#0000ff",
    },
  ];

  return (
    <div style={{ width: "100%", height: "400px" }}>
      <Map ncpKeyId="your-ncp-key-id">
        {areas.map((area, index) => (
          <Polygon
            key={index}
            paths={area.paths}
            fillColor={area.fillColor}
            fillOpacity={0.3}
            strokeColor={area.fillColor}
            strokeWeight={2}
          />
        ))}
      </Map>
    </div>
  );
}

export default MultiplePolygons;
```
