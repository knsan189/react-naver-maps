# 도형 그리기 예제

지도 클릭으로 점을 추가해 폴리라인과 폴리곤을 직접 그리는 예제입니다.

## 라이브 예제

<BrowserOnly>{() => <ShapeDrawingExample />}</BrowserOnly>

## 코드

```tsx
import { useState } from "react";
import { Map, Marker, Polygon, Polyline } from "@rousen/react-naver-maps";

function ShapeDrawing() {
  const [points, setPoints] = useState<Array<{ x: number; y: number }>>([]);

  const handleMapClick = (event: naver.maps.PointerEvent) => {
    const { x, y } = event.coord;
    setPoints((prev) => [...prev, { x, y }]);
  };

  const handleUndo = () => {
    setPoints((prev) => prev.slice(0, -1));
  };

  const handleClear = () => {
    setPoints([]);
  };

  return (
    <div style={{ margin: "20px 0" }}>
      <div style={{ marginBottom: "10px" }}>
        <button onClick={handleUndo} disabled={points.length === 0}>
          마지막 점 삭제
        </button>
        <button
          onClick={handleClear}
          disabled={points.length === 0}
          style={{ marginLeft: "6px" }}
        >
          초기화
        </button>
        <span style={{ marginLeft: "10px", fontSize: "14px" }}>
          점 {points.length}개
        </span>
      </div>
      <div style={{ width: "100%", height: "400px" }}>
        <Map
          ncpKeyId="your-ncp-key-id"
          mapOptions={{ center: { x: 127.0276, y: 37.4979 }, zoom: 13 }}
          onClick={handleMapClick}
        >
          {points.map((point, index) => (
            <Marker key={index} position={point} />
          ))}
          {points.length >= 2 && (
            <Polyline
              path={points}
              strokeColor="#2563eb"
              strokeOpacity={0.9}
              strokeWeight={3}
            />
          )}
          {points.length >= 3 && (
            <Polygon
              paths={[points]}
              fillColor="#60a5fa"
              fillOpacity={0.3}
              strokeColor="#2563eb"
              strokeOpacity={0.9}
              strokeWeight={2}
            />
          )}
        </Map>
      </div>
    </div>
  );
}

export default ShapeDrawing;
```

## 설명

- 지도 클릭으로 점을 추가하면 폴리라인이 그려집니다
- 점이 3개 이상이면 폴리곤이 자동으로 채워집니다
- 버튼으로 마지막 점 삭제/전체 초기화를 할 수 있습니다
